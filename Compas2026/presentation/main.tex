\begin{frame}{Gysela --- Simulation gyrocinétique de la turbulence plasma}

  \vspace{0.4em}

  \begin{columns}[t]
    \begin{column}{0.48\textwidth}
      \begin{block}{Contexte : fusion nucléaire}
        \begin{itemize}
          \item Les tokamaks confinent un plasma à très haute température
          \item La \textbf{turbulence plasma} dégrade le confinement
          \item \textbf{Gysela} : code de simulation gyrocinétique développé par le CEA pour étudier cette turbulence
          \item Modèle de Vlasov gyrocinétique \textbf{5D} :
                $f(r,\ \theta,\ \varphi,\ v_\parallel,\ \mu)$
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{L'écriture des checkpoints présente un problème}
        \begin{itemize}
          \item Les checkpoints en production sont volumineux et \textbf{bloquent} la simulation
          \item Variabilité jusqu'à \textbf{$\times$3} sur un système de fichier parallel (PFS) partagé en production
          \begin{itemize}
            \item Trouver une meilleure stratégie I/O est \textbf{critique}
          \end{itemize}
          \item Volume de données très important à écrire
        \end{itemize}
      \end{alertblock}

      \vspace{0.4em}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Problématique}
  \begin{block}{Notre objectif}
    \begin{itemize}
      \item \textbf{Améliorer} les performances d'écriture des checkpoints
      \item \textbf{Étudier} l'impact de plusieurs paramètres I/O
      \item Trouver la meilleure stratégie I/O pour Gysela en contexte HPC
    \end{itemize}
  \end{block}
\end{frame}

\section{Concepts clés}
\frame{\sectionpage}
\begin{frame}{Système de fichiers parallèles -- PFS}
  \begin{columns}[t]
    
    \begin{column}{0.46\textwidth}
      \begin{block}{Object Storage Target -- OST}
        \begin{itemize}
          \item Unité de stockage logique ou physique
          \begin{itemize}
            \item Serveur de stockage
            \item groupe de disques RAID
          \end{itemize}
          \item Stockage des données sous forme d'objets
          \item Permet un découpage des données sur plusieurs système de stockage 
        \end{itemize}
      \end{block}
    \end{column}

    \begin{column}{0.46\textwidth}
      \begin{block}{Le striping}
        \begin{itemize}
          \item Technique de distribution de données
          \begin{itemize}
            \item Découpage d'un fichier en blocs (\textit{stripes})
            \item Repartition de ces blocs en parallèle sur les OST
          \end{itemize}
          \item Paramètre clés configurables:
          \begin{itemize}
            \item \textbf{Stripe Count}: nombre d'OST utilisés (ex: 1 à 8)
            \item \textbf{Stripe Size}: taille de chaque bloc de données écrit dans un \textit{round-robin}
          \end{itemize}
          \item Objectif : Augmenter l'usage de la bande passante
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Single Shared File --- SSF}
  \begin{columns}[t]
    \begin{column}{0.30\textwidth}      
        \begin{itemize}
          \item Tous les rang MPI écrivent en parallèle dans un fichier commun.
          \item Accès non contigus calculés par HDF5/MPI-IO
          \item Aggrège toutes les données avant de les envoyer au PFS
        \end{itemize}
    \end{column}

    \begin{column}{0.60\textwidth}
      % Figure explication SSF %
      \begin{center}
        \input{figures/ssf-figure.tex}
      \end{center}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Subfiling}
  \begin{columns}[t]
    \begin{column}{0.30\textwidth}
      \begin{itemize}
        \item Les rangs MPI sont divisés en plusieurs \textbf{groupes (sous-ensembles)}.
        \item Chaque groupe élit un ou plusieurs \textbf{I/O Concentrators (IOC)} / Agrégateurs.
        \item L'application écrit un nombre fixe de fichiers intermédiaire régient par un fichier de configuration
      \end{itemize}
    \end{column}

    \begin{column}{0.60\textwidth}
      \begin{center}
        \input{figures/subfiling-figure.tex}
      \end{center}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{File Per Process --- FPP}
  \begin{columns}[t]
    \begin{column}{0.30\textwidth}
      \begin{itemize}
        \item Chaque rang MPI gère son unique portion de données
        \item Chaque rang crée, écrit et ferme son fichier indépendamment des autres
        \item Génère autant de fichier que de processus MPI
      \end{itemize}
    \end{column}

    \begin{column}{0.60\textwidth}
      \begin{center}
        \input{figures/fpp-figure.tex}
      \end{center}
    \end{column}
  \end{columns}
\end{frame}

        \begin{itemize}

      \end{block}

      \begin{exampleblock}{Plafrim --- Première étude de cas}
        \begin{itemize}

      \end{exampleblock}

% --- Subfiling ---
\begin{frame}{Subfiling --- Nombre optimal de subfiles et modification du code}

  \vspace{0.5em}

  \begin{columns}[t]
    \begin{column}{0.55\textwidth}
      \begin{center}
        \includegraphics[width=0.95\textwidth]{figures/fig4_subfile_count_talk.pdf}
        \textit{[Figure : temps d'écriture (s) vs nombre de subfiles, config 16$\times$8]}
      \end{center}
    \end{column}
    \begin{column}{0.40\textwidth}

      \vspace{1em}

      \begin{block}{Observation}
        \begin{itemize}
          \item Optimal à \textbf{8 subfiles} = nb OST
          \item Confirme : \textbf{nb\_subfiles = nb\_OST}
          \item Différence 8 vs 16 : $<$3\%
        \end{itemize}
      \end{block}

      \vspace{0.8em}

      \begin{exampleblock}{Modification du code requise}
        Le Subfiling nécessite \texttt{MPI\_THREAD\_MULTIPLE}\\[4pt]
        \texttt{MPI\_Init()} $\rightarrow$ \texttt{MPI\_Init\_thread(...,}\\
        \texttt{MPI\_THREAD\_MULTIPLE, ...)}\\[4pt]
        Nécessaire pour les appels MPI des threads IOC
      \end{exampleblock}

    \end{column}
  \end{columns}

\end{frame}

% --- FPP ---
\begin{frame}{FPP --- Bonnes performances constantes}

  \vspace{0.5em}

  \begin{columns}[t]
    \begin{column}{0.55\textwidth}
      \begin{center}
        \includegraphics[width=0.95\textwidth]{figures/fig5_correct.pdf}
        \textit{[Figure : temps d'écriture et temps total (s) vs nœuds --- FPP uniquement]}
      \end{center}
    \end{column}
    \begin{column}{0.40\textwidth}


      \begin{block}{Performances peak}
        Meilleures performances I/O avec une très légère dégradation qui se stabilise à 8 nœuds
      \end{block}


      \begin{alertblock}{Le vrai compromis}
        FPP ne nécessite pas de topologie particulière pour ses meilleures performances I/O.\\[4pt]
        Gysela en production requiert \textbf{beaucoup de nœuds} -- heureusement les performances restent stables entre petites et grandes configurations.
      \end{alertblock}
    \end{column}
  \end{columns}

\end{frame}
